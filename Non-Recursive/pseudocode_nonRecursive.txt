Algorithm findDominator(A, N)
// A is an array of size N
{
    candidate := A[0]
    count     := 1

    for i ← 1 to N - 1 step 1 do
    {
        if count = 0 then
        {
            candidate := A[i]
            count     := 1
        }
        else if A[i] = candidate then
            count := count + 1
        else
            count := count - 1
    }

    occurrences := 0
    for i ← 0 to N - 1 step 1 do
    {
        if A[i] = candidate then
            occurrences := occurrences + 1
    }

    if occurrences > N / 2 then
        return candidate
    else
        return -1
}

Algorithm main()
{
    read N

    if N = 0 then
    {
        write "No dominator found."
        return
    }

    for i ← 0 to N - 1 step 1 do
    {
        read A[i]
    }

    dominator := findDominator(A, N)

    if dominator != -1 then
    {
        write "Dominator value : ", dominator
        write "Found at indices: "
        for i ← 0 to N - 1 step 1 do
        {
            if A[i] = dominator then
                write i
        }
    }
    else
        write -1
}
