Algorithm countOccurrences(A, index, N, cand)
// A is an array of size N
{
    if index = N then
        return 0
    if A[index] = cand then
        return 1 + countOccurrences(A, index + 1, N, cand)
    else
        return countOccurrences(A, index + 1, N, cand)
}

Algorithm findDominator(A, i, N, cand, cnt)
// A is an array of size N
{
    if i = N then
    {
        occurrences := countOccurrences(A, 0, N, cand)
        if occurrences > N / 2 then
            return cand
        else
            return -1
    }
    if cnt = 0 then
    {
        cand := A[i]
        cnt  := 1
    }
    else if A[i] = cand then
        cnt := cnt + 1
    else
        cnt := cnt - 1
    return findDominator(A, i + 1, N, cand, cnt)
}

Algorithm main()
{
    read N
    for i ← 0 to N - 1 step 1 do
    {
        read A[i]
    }
    cand := 0
    cnt  := 0
    dominator := findDominator(A, 0, N, cand, cnt)
    if dominator != -1 then
    {
        write "Dominator = ", dominator
        write "Indices: "
        for i ← 0 to N - 1 step 1 do
        {
            if A[i] = dominator then
                write i
        }
    }
    else
    {
        write dominator
        write "No dominator"
    }
}
